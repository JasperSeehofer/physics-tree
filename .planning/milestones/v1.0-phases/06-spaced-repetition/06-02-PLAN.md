       ```rust
       }

       #[derive(Deserialize)]
       pub struct SkipReviewRequest {
           pub node_id: Uuid,
       }

       #[derive(Serialize)]
       pub struct DueCountResponse {
           pub due_count: i64,
       }

       #[derive(Serialize)]
       pub struct SuggestionsResponse {
           pub suggestions: Vec<db::review_repo::FrontierSuggestion>,
       }
       ```

       **GET /api/review/queue handler (`get_review_queue`):**
       - Extract session user_id (401 if missing)
       - Call db::review_repo::get_review_queue(pool, user_id)

       - Extract session user_id (401 if missing)

       - Return SubmitReviewResponse with xp_awarded, rating, next_review_date (formatted as ISO 8601), streak, freeze_tokens, freeze_used

       **POST /api/review/skip handler (`skip_review`):**
       - Extract session user_id (401 if missing)
       - Call db::review_repo::skip_review(pool, user_id, req.node_id)
       - Return 200 OK

       **GET /api/review/due-count handler (`get_due_count`):**
       - Extract session user_id (401 if missing)
       - Call db::review_repo::get_due_count(pool, user_id)
       - Return DueCountResponse { due_count }

       **GET /api/review/suggestions handler (`get_suggestions`):**
       - Extract session user_id (401 if missing)
       - Call db::review_repo::get_frontier_suggestions(pool, user_id, 3)
       - Return SuggestionsResponse { suggestions }
