**2. SimulationEmbed (`crates/app/src/components/simulation/embed.rs`):**

This component manages the canvas, the rAF loop, instantiates the simulation, and syncs state to/from URL query params (per D-15).

Props:
- `sim_name: String` — which simulation to load (e.g. "projectile")
- `expanded: RwSignal<bool>` — whether in breakout (wider) mode

State signals:
- `running: RwSignal<bool>` — play/pause state
- `angle: RwSignal<f32>` — current angle (for projectile; generalize later)
- `speed: RwSignal<f32>` — current speed
- `show_plot: RwSignal<bool>` — plot visibility toggle
- `precise_mode: RwSignal<bool>` — toggle for numeric input fields (per D-11)

**URL state sync (per D-15):**
On component mount:
1. Read `window.location.search` via `web_sys::Url::new(&web_sys::window().unwrap().location().href().unwrap())`
2. Parse URL search params: `url.search_params()` to read `angle`, `speed`, and any other sim-specific params
3. If params exist, initialize signals from them instead of defaults
4. Example: `?angle=60&speed=30` initializes angle=60.0, speed=30.0

On parameter change (angle/speed signal effects):
1. Build new URL search params string from current signal values
2. Use `window.history().replace_state_with_url(...)` to update URL without navigation
3. Only update params relevant to the current simulation (e.g., projectile uses `angle` and `speed`)

This enables sharing specific simulation setups via URL.

Component structure:
```rust
#[component]
pub fn SimulationEmbed(sim_name: String) -> impl IntoView {
    let canvas_ref = NodeRef::<leptos::html::Canvas>::new();
    let expanded = RwSignal::new(false);
    let precise_mode = RwSignal::new(false);
    let show_plot = RwSignal::new(false);

    // Initialize from URL params or defaults
    let (initial_angle, initial_speed) = read_url_params_for_sim(&sim_name);
    let running = RwSignal::new(false);
    let angle = RwSignal::new(initial_angle);
    let speed = RwSignal::new(initial_speed);

    // URL sync effect: on param change, update URL
    #[cfg(target_arch = "wasm32")]
    Effect::new(move |_| {
        let a = angle.get();
        let s = speed.get();
        update_url_params(&sim_name_clone, &[("angle", a), ("speed", s)]);
    });

    // ... rAF loop setup same as before ...

    view! {
        <div class=container_class>
            <div class="relative bg-bark-mid rounded-lg overflow-hidden">
                <canvas node_ref=canvas_ref class="w-full" style="aspect-ratio: 16/9" />
                // Play button overlay when paused (per D-12)
            </div>
            <SimulationControls
                running=running angle=angle speed=speed
                show_plot=show_plot expanded=expanded
                precise_mode=precise_mode
                sim_name=sim_name.clone()
            />
            <Show when=move || show_plot.get()>
                <SimulationPlot />
            </Show>
        </div>
    }
}

// Helper: read URL search params for simulation
#[cfg(target_arch = "wasm32")]
fn read_url_params_for_sim(sim_name: &str) -> (f32, f32) {
    let window = web_sys::window().unwrap();
    let url = web_sys::Url::new(&window.location().href().unwrap()).unwrap();
    let params = url.search_params();
    let angle = params.get("angle").and_then(|v| v.parse().ok()).unwrap_or(45.0);
    let speed = params.get("speed").and_then(|v| v.parse().ok()).unwrap_or(20.0);
    (angle, speed)
}

// Helper: update URL params without navigation
#[cfg(target_arch = "wasm32")]
fn update_url_params(sim_name: &str, params: &[(&str, f32)]) {
    let window = web_sys::window().unwrap();
    let url = web_sys::Url::new(&window.location().href().unwrap()).unwrap();
    let search_params = url.search_params();
    for (key, val) in params {
        search_params.set(key, &format!("{:.1}", val));
    }
    let _ = window.history().unwrap().replace_state_with_url(
        &wasm_bindgen::JsValue::NULL, "", Some(&url.href())
    );
}
```

import { parse, simplify, evaluate } from 'mathjs';

window.__mathjs_bridge = {
    checkEquivalence(userExpr, expectedExpr, variablesJson) {
        try {
            const variables = JSON.parse(variablesJson);
            const userParsed = parse(userExpr);
            const expectedParsed = parse(expectedExpr);
            const samplePoints = 5;
            for (let i = 0; i < samplePoints; i++) {
                const scope = {};
                for (const v of variables) {
                    scope[v] = Math.random() * 10 + 0.1;
                }
                const userVal = userParsed.evaluate(scope);
                const expectedVal = expectedParsed.evaluate(scope);
                if (Math.abs(userVal - expectedVal) > 1e-6) return false;
            }
            return true;
        } catch (e) {
            return false;
        }
    }
};
```

pub async fn get_quiz(
    Path(slug): Path<String>,
) -> Result<Json<Vec<QuizQuestion>>, (StatusCode, String)> {
    // Read quiz JSON file from content/{branch}/{slug}.quiz.json
    let quiz_path = format!("content/classical-mechanics/{}.quiz.json", slug);
    let quiz_json = tokio::fs::read_to_string(&quiz_path)
        .await
        .map_err(|_| (StatusCode::NOT_FOUND, "Quiz not found".into()))?;
    let mut questions: Vec<QuizQuestion> = serde_json::from_str(&quiz_json)
        .map_err(|e| (StatusCode::INTERNAL_SERVER_ERROR, e.to_string()))?;

    // Randomize per D-22: pick 4-5 from pool, shuffle options
    use rand::seq::SliceRandom;
    let mut rng = rand::thread_rng();
    questions.shuffle(&mut rng);
    questions.truncate(5);
    // Shuffle multiple choice options
    for q in &mut questions {
        if let Some(ref mut opts) = q.options {
            opts.shuffle(&mut rng);
        }
    }
    Ok(Json(questions))
}
```
