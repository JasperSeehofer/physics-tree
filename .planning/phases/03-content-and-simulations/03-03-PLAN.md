**2. SimulationEmbed (`crates/app/src/components/simulation/embed.rs`):**

This component manages the canvas, the rAF loop, and instantiates the simulation.

Props:
- `sim_name: String` — which simulation to load (e.g. "projectile")
- `expanded: RwSignal<bool>` — whether in breakout (wider) mode

State signals:
- `running: RwSignal<bool>` — play/pause state
- `angle: RwSignal<f32>` — current angle (for projectile; generalize later)
- `speed: RwSignal<f32>` — current speed
- `show_plot: RwSignal<bool>` — plot visibility toggle

Component structure:
```rust
#[component]
pub fn SimulationEmbed(sim_name: String) -> impl IntoView {
    let canvas_ref = NodeRef::<leptos::html::Canvas>::new();
    let running = RwSignal::new(false);
    let angle = RwSignal::new(45.0f32);
    let speed = RwSignal::new(20.0f32);
    let show_plot = RwSignal::new(false);
    let expanded = RwSignal::new(false);

    // Container class changes based on expanded state
    // Default: within 700px content column
    // Expanded: max-w-[960px] mx-auto (per UI-SPEC)

    #[cfg(target_arch = "wasm32")]
    {
        use std::rc::Rc;
        use std::cell::RefCell;
        use wasm_bindgen::prelude::*;
        use wasm_bindgen::closure::Closure;

        Effect::new(move |_| {
            if let Some(canvas) = canvas_ref.get() {
                let canvas_el: web_sys::HtmlCanvasElement = canvas.clone().into();
                canvas_el.set_width(960);
                canvas_el.set_height(540);

                // Create simulation via wasm-bindgen
                let sim = Rc::new(RefCell::new(
                    simulation::WasmProjectile::new(960.0, 540.0)
                ));

                // Initial render (static preview per D-12)
                {
                    let s = sim.borrow();
                    s.tick(&canvas_el); // renders initial state
                }

                // rAF loop
                let sim_clone = sim.clone();
                let canvas_clone = canvas_el.clone();
                let raf_id = Rc::new(RefCell::new(0i32));
                let raf_id_clone = raf_id.clone();
                let closure: Rc<RefCell<Option<Closure<dyn FnMut()>>>> = Rc::new(RefCell::new(None));
                let closure_clone = closure.clone();

                let cb = Closure::new(move || {
                    let mut s = sim_clone.borrow_mut();
                    s.tick(&canvas_clone);
                    if s.is_running() {
                        let window = web_sys::window().unwrap();
                        let id = window.request_animation_frame(
                            closure_clone.borrow().as_ref().unwrap().as_ref().unchecked_ref()
                        ).unwrap();
                        *raf_id_clone.borrow_mut() = id;
                    }
                });
                *closure.borrow_mut() = Some(cb);

                // Store sim for controls to interact with
                // Use provide_context or a shared Rc
                // Controls call sim.borrow_mut().set_angle() etc.

                on_cleanup(move || {
                    let window = web_sys::window().unwrap();
                    window.cancel_animation_frame(*raf_id.borrow()).ok();
                    closure.borrow_mut().take(); // drop the closure
                });
            }
        });
    }

    let container_class = move || {
        if expanded.get() {
            "max-w-[960px] mx-auto"
        } else {
            "w-full"
        }
    };

    view! {
        <div class=container_class>
            <div class="relative bg-bark-mid rounded-lg overflow-hidden">
                <canvas
                    node_ref=canvas_ref
                    class="w-full"
                    style="aspect-ratio: 16/9"
                />
                // Play button overlay when paused (per D-12)
                <Show when=move || !running.get()>
                    <button
                        class="absolute inset-0 flex items-center justify-center"
                        aria-label="Play simulation"
                        on:click=move |_| { /* sim.play(); start rAF loop */ running.set(true); }
                    >
                        <div class="w-12 h-12 rounded-full bg-leaf-green flex items-center justify-center">
                            // Play triangle SVG
                        </div>
                    </button>
                </Show>
            </div>
            <SimulationControls
                running=running
                angle=angle
                speed=speed
                show_plot=show_plot
                expanded=expanded
                sim_name=sim_name.clone()
            />
            <Show when=move || show_plot.get()>
                <SimulationPlot />
            </Show>
        </div>
    }
}
```

The key challenge: sharing the `Rc<RefCell<WasmProjectile>>` between the rAF loop and the controls. Use `provide_context` to share a wrapper type, or pass callback closures to the controls component.

**3. mathjs bridge (`crates/app/src/js/mathjs_bridge.js`):**
```javascript
import { parse, simplify, evaluate } from 'mathjs';

window.__mathjs_bridge = {
    checkEquivalence(userExpr, expectedExpr, variablesJson) {
        try {
            const variables = JSON.parse(variablesJson);
            const userParsed = parse(userExpr);
            const expectedParsed = parse(expectedExpr);
            const samplePoints = 5;
            for (let i = 0; i < samplePoints; i++) {
                const scope = {};
                for (const v of variables) {
                    scope[v] = Math.random() * 10 + 0.1;
                }
                const userVal = userParsed.evaluate(scope);
                const expectedVal = expectedParsed.evaluate(scope);
                if (Math.abs(userVal - expectedVal) > 1e-6) return false;
            }
            return true;
        } catch (e) {
            return false;
        }
    }
};
```

Bundle via esbuild into `/js/mathjs_bundle.js`. Add script tag to shell in `crates/app/src/lib.rs`. Use lazy loading: only load this bundle when a formula quiz is present (use dynamic `<script>` injection on concept pages that have formula questions, rather than loading on all pages).

**7. QuizMatching (`crates/app/src/components/quiz/matching.rs`):**

- Display pairs as two columns: left items, right items (shuffled)
- User clicks a left item, then clicks a right item to form a pair

- When all pairs matched, "Check answer" button appears

pub async fn get_quiz(
    Path(slug): Path<String>,
) -> Result<Json<Vec<QuizQuestion>>, (StatusCode, String)> {
    // Read quiz JSON file from content/{branch}/{slug}.quiz.json
    let quiz_path = format!("content/classical-mechanics/{}.quiz.json", slug);
    let quiz_json = tokio::fs::read_to_string(&quiz_path)
        .await
        .map_err(|_| (StatusCode::NOT_FOUND, "Quiz not found".into()))?;
    let mut questions: Vec<QuizQuestion> = serde_json::from_str(&quiz_json)
        .map_err(|e| (StatusCode::INTERNAL_SERVER_ERROR, e.to_string()))?;

    // Randomize per D-22: pick 4-5 from pool, shuffle options
    use rand::seq::SliceRandom;
    let mut rng = rand::thread_rng();
    questions.shuffle(&mut rng);
    questions.truncate(5);
    // Shuffle multiple choice options
    for q in &mut questions {
        if let Some(ref mut opts) = q.options {
            opts.shuffle(&mut rng);
        }
    }
    Ok(Json(questions))
}
