**1. Implement `crates/db/src/graph_repo.rs`** with three public async functions:

```rust
use domain::{PhysicsNode, PhysicsEdge, NodeType, EdgeType};
use sqlx::PgPool;
use uuid::Uuid;

pub async fn get_all_nodes(pool: &PgPool) -> Result<Vec<PhysicsNode>, sqlx::Error> {
    sqlx::query_as!(
        PhysicsNode,
        r#"SELECT id, slug, title,
                  node_type AS "node_type: NodeType",
                  branch, depth_tier, description
           FROM nodes
           ORDER BY branch, depth_tier"#
    )
    .fetch_all(pool)
    .await
}

pub async fn get_all_edges(pool: &PgPool) -> Result<Vec<PhysicsEdge>, sqlx::Error> {
    sqlx::query_as!(
        PhysicsEdge,
        r#"SELECT from_node, to_node,
                  edge_type AS "edge_type: EdgeType",
                  weight
           FROM edges"#
    )
    .fetch_all(pool)
    .await
}

pub async fn get_prereq_chain(
    pool: &PgPool,
    node_id: Uuid,
) -> Result<Vec<PhysicsNode>, sqlx::Error> {
    sqlx::query_as!(
        PhysicsNode,
        r#"WITH RECURSIVE prereqs AS (
               SELECT from_node FROM edges
               WHERE to_node = $1 AND edge_type = 'prerequisite'
               UNION
               SELECT e.from_node FROM edges e
               JOIN prereqs p ON e.to_node = p.from_node
               WHERE e.edge_type = 'prerequisite'
           )
           SELECT id, slug, title,
                  node_type AS "node_type: NodeType",
                  branch, depth_tier, description
           FROM nodes WHERE id IN (SELECT from_node FROM prereqs)"#,
        node_id
    )
    .fetch_all(pool)
    .await
}
```

Important: The `query_as!` macro requires exact column type annotations with the `AS "col: Type"` syntax for enum columns. The `NodeType` and `EdgeType` types must be imported from `domain` crate.

use axum::{extract::{Path, State}, Json};
use sqlx::PgPool;
use uuid::Uuid;
use domain::{PhysicsNode, PhysicsEdge};

#[derive(serde::Serialize)]
pub struct GraphData {
    pub nodes: Vec<PhysicsNode>,
    pub edges: Vec<PhysicsEdge>,
}

pub async fn get_graph(
    State(pool): State<PgPool>,
) -> Result<Json<GraphData>, (axum::http::StatusCode, String)> {
    let nodes = db::graph_repo::get_all_nodes(&pool).await
        .map_err(|e| (axum::http::StatusCode::INTERNAL_SERVER_ERROR, e.to_string()))?;
    let edges = db::graph_repo::get_all_edges(&pool).await
        .map_err(|e| (axum::http::StatusCode::INTERNAL_SERVER_ERROR, e.to_string()))?;
    Ok(Json(GraphData { nodes, edges }))

pub async fn get_prereqs(
    State(pool): State<PgPool>,
    Path(node_id): Path<Uuid>,

        .map_err(|e| (axum::http::StatusCode::INTERNAL_SERVER_ERROR, e.to_string()))?;
    Ok(Json(prereqs))
}

- Read `DATABASE_URL` from environment (use `std::env::var("DATABASE_URL")`)
- Call `db::create_pool(&database_url).await.unwrap()` to create PgPool

The key change in main.rs: the current `.with_state(leptos_options)` must become a merged state or the API routes must be nested with their own state. The cleanest approach:

```rust
let pool = db::create_pool(&std::env::var("DATABASE_URL")
    .expect("DATABASE_URL must be set")).await.unwrap();

let app = api_routes(pool.clone())
    .leptos_routes(&leptos_options, routes, {
        let leptos_options = leptos_options.clone();
        move || app::shell(leptos_options.clone())
    })
    .fallback(leptos_axum::file_and_error_handler(app::shell))
    .with_state(leptos_options);
```

Wait — this won't work because `api_routes` returns a `Router` with `PgPool` state while `.with_state(leptos_options)` sets `LeptosOptions` state. Use `.merge()` or nest the API router. The correct pattern for Axum 0.8 with mixed state:

```rust
let api = api_routes(pool.clone()); // Router with PgPool state, already called .with_state(pool) inside

let app = Router::new()
    .merge(api) // API routes with their own state — merge converts to Router<()>
    .leptos_routes(&leptos_options, routes, { ... })
    .fallback(leptos_axum::file_and_error_handler(app::shell))
    .with_state(leptos_options);
```

This works because `api_routes` calls `.with_state(pool)` internally, which converts it to `Router<()>`. Merging a `Router<()>` into another `Router<_>` is always valid.
