    ```rust
    pub fn compute_xp(depth_tier: &str, score_pct: u32, hints_used: bool) -> u32 {
        if score_pct < 70 { return 0; }
        let base: u32 = match depth_tier {
            "root" => 15, "trunk" => 20, "branch" => 30, "leaf" => 40, _ => 20,
        };
        if score_pct == 100 && !hints_used {
            (base as f64 * 1.5).round() as u32
        } else if hints_used {
            (base as f64 * score_pct as f64 / 100.0 * 0.5).round() as u32
        } else {
            (base as f64 * score_pct as f64 / 100.0).round() as u32
        }
    }
    ```

    Exact implementation:
    ```rust
    #[cfg(not(target_arch = "wasm32"))]
    pub fn extract_latex_placeholders(input: &str) -> String {
        use regex::Regex;
        let display_re = Regex::new(r"\$\$([\s\S]*?)\$\$").unwrap();
        let s = display_re.replace_all(input, |caps: &regex::Captures| {
            let latex = html_attr_escape(caps[1].trim());
            format!(r#"<div data-latex="{latex}" data-display="true"></div>"#)
        });
        let inline_re = Regex::new(r"\$([^$\n]+)\$").unwrap();
        inline_re.replace_all(&s, |caps: &regex::Captures| {
            let latex = html_attr_escape(caps[1].trim());
            format!(r#"<span data-latex="{latex}" data-display="false"></span>"#)
        }).to_string()
    }
    ```
