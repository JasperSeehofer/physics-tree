```rust
use leptos::prelude::*;

/// Health check status indicator pill.
/// Fetches /api/health on mount and displays system status.
#[component]
pub fn HealthIndicator() -> impl IntoView {
    let health_status = Resource::new(
        || (),
        |_| async move {
            // Use gloo-net or raw fetch for WASM-compatible HTTP
            // For SSR, this runs server-side
            #[cfg(feature = "hydrate")]
            {
                let resp = gloo_net::http::Request::get("/api/health")
                    .send()
                    .await;
                match resp {
                    Ok(r) => r.ok(),
                    Err(_) => false,
                }
            }
            #[cfg(not(feature = "hydrate"))]
            {
                // During SSR, assume healthy (client will re-check on hydration)
                true
            }
        },
    );

    view! {
        <Suspense fallback=move || view! {
            <div class="px-4 py-2 rounded-node text-sm font-normal bg-bark-mid text-mist">
                "Checking system\u{2026}"
            </div>
        }>
            {move || health_status.get().map(|ok| {
                if ok {
                    view! {
                        <div class="px-4 py-2 rounded-node text-sm font-normal bg-leaf-green/20 text-leaf-green">
                            "System operational"
                        </div>
                    }
                } else {
                    view! {
                        <div class="px-4 py-2 rounded-node text-sm font-normal bg-bloom-pink/20 text-bloom-pink">
                            "System unavailable"
                        </div>
                    }
                }
            })}
        </Suspense>
    }
}
```

Note: Add `gloo-net` to `crates/app/Cargo.toml` as a dependency for WASM-compatible HTTP requests:
```toml
[dependencies]
gloo-net = { version = "0.6", features = ["http"] }
```
Only add it under the hydrate feature if needed, but gloo-net compiles for both targets. If gloo-net causes issues, fall back to `web_sys::window().unwrap().fetch_with_str()` pattern. The key requirement is that the component compiles for both SSR and WASM targets.

**Alternative approach if gloo-net is problematic:** Use Leptos server functions instead:
```rust
#[server(HealthCheck, "/api")]
pub async fn check_health() -> Result<bool, ServerFnError> {
    Ok(true) // Server-side health check
}
```
This is simpler and avoids cross-target HTTP issues. Use this approach if the gloo-net approach causes compilation problems.

```rust
use leptos::prelude::*;
use pages::landing::LandingPage;

/// Shell function for Leptos SSR — provides the HTML document wrapper.
/// Called by the server to generate the initial HTML.
pub fn shell(options: LeptosOptions) -> impl IntoView {
    view! {
        <!DOCTYPE html>
        <html lang="en" class="dark">
            <head>
                <meta charset="utf-8" />
                <meta name="viewport" content="width=device-width, initial-scale=1" />
                <title>"PhysicsTree \u{2014} Explore the physics universe"</title>
                <AutoReload options=options.clone() />
                <HydrationScripts options />
                <leptos_meta::MetaTags />
            </head>
            <body>
                <App />
            </body>
        </html>
    }
}

/// Root application component.
#[component]
pub fn App() -> impl IntoView {
    view! {
        <LandingPage />
    }
}
```
